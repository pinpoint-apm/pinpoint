### Pinpoint Configuration
pinpoint.config

#### Set enable options.
~~~
###########################################################
# Reactor
###########################################################
profiler.reactor.enable=true
# Set whether to trace the Subscriber.onError(Throwable t) method
profiler.reactor.trace.subscribe.error=true
~~~
